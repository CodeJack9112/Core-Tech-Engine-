Put these lines into CMakeLists.txt in a new optional section guarded by an option:
option(ENABLE_PROFILER "Enable the optional profiler overlay" OFF)
if(ENABLE_PROFILER)
  add_library(ProfilerOverlay tools/quick_profiler/ProfilerOverlay.cpp)
  target_include_directories(ProfilerOverlay PUBLIC ${PROJECT_SOURCE_DIR}/third_party/imgui)
  target_link_libraries(${PROJECT_NAME} PRIVATE ProfilerOverlay)
endif()
